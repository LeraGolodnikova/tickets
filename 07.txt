CREATE DATABASE tick5_07;

CREATE TABLE edditional_conditions (
	id SERIAL PRIMARY KEY,
	has_housing BOOLEAN NOT NULL,
	has_children BOOLEAN NOT NULL,
	no_bab_habits BOOLEAN NOT NULL
);

CREATE TABLE requirements (
	id SERIAL PRIMARY KEY,
	education VARCHAR(50),
	age_min INT CHECK(age_min > 18 AND age_min < 100),
	age_max INT CHECK(age_max >= age_min AND age_max < 100),
	nationality VARCHAR(50),
	monthly_income NUMERIC (10,2),
	edditional_conditions_id INT REFERENCES edditional_conditions(id)
);

CREATE TABLE client (
	id SERIAL PRIMARY KEY,
	last_name VARCHAR(50) NOT NULL,
	first_name VARCHAR(100) NOT NULL,
	middle_name VARCHAR(100),
	age INT NOT NULL CHECK(age >18 AND age< 100),
	nationality VARCHAR(50) NOT NULL,
	education VARCHAR(50) NOT NULL,
	monthly_income NUMERIC (10,2) NOT NULL,
	edditional_conditions_id INT NOT NULL REFERENCES edditional_conditions(id),
	requirements_id INT NOT NULL REFERENCES requirements(id)
);

SELECT c.id, c.last_name, c.first_name, c.age
FROM client c
JOIN requirements r ON requirements_id = r.id
JOIN edditional_conditions ec ON r.edditional_conditions_id = ec.id
WHERE no_bab_habits = TRUE;

--Высшее образование, имеют жил площадь и не старше 30
SELECT c.id, c.last_name, c.first_name, c.age, c.education, ec.has_housing
FROM client c
JOIN edditional_conditions ec ON  edditional_conditions_id = ec.id
WHERE 
	c.education = 'Высшее'
	AND ec.has_housing = TRUE
	AND c.age <=30;
